func verify(name:String,age:Int) -> String{
    var text = ("\(name) is \(age) years old")
    if(age < 18){
        text += "，不可以看片"

//获取列表里的最大最小值
func minMax(list:[Int]) -> (min:Int,max:Int){   //传入参数list为int列表，返回值为元组
    var min = list[0],max = list[0]     //初始化最小值和最大值为列表第一个元素
    for num in list{
        if num < min{
            min = num
        }else if num > max{
            max = num
        }
    }
    return (min,max)
}
var l = [31,3125,532,21,3]
var b = minMax(list: l)
print("最小值为\(b.min),最大值为\(b.max)")

//函数嵌套,外部的函数可以调用函数内定义的函数
func tools(num1:Int,num2:Int) ->() -> Int{
    var num = 0
    func add()->Int{
        num = num1 + num2
        return num
    }
    return add
}

var f = tools(num1: 33, num2: 22)
print(f())

//类
class Player{
    var score = 0
    init(score:Int){
        self.score = score
    }
    func playerLevel() -> String{
        if self.score < 10{
                return "新手玩家"
        }else if self.score < 20{
            return "普通玩家"
        } else if self.score < 30{
            return "大师玩家"
        }else{
            return "游客"
        }
    }
}

var mike = Player(score: 15)
print(mike.playerLevel())
