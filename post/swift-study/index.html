//获取列表里的最大最小值
func minMax(list:[Int]) -> (min:Int,max:Int){   //传入参数list为int列表，返回值为元组
    var min = list[0],max = list[0]     //初始化最小值和最大值为列表第一个元素
    for num in list{
        if num < min{
            min = num
        }else if num > max{
            max = num
        }
    }
    return (min,max)
}
var l = [31,3125,532,21,3]
var b = minMax(list: l)
print("最小值为\(b.min),最大值为\(b.max)")

//函数嵌套,外部的函数可以调用函数内定义的函数
func tools(num1:Int,num2:Int) ->() -> Int{
    var num = 0
    func add()->Int{
        num = num1 + num2
        return num
    }
    return add
}

var f = tools(num1: 33, num2: 22)
print(f())

//定义协议（类似Java的接口）
protocol Teacher{
    var teacherName:String {get}
    var subjects:[String]{get set}
    func teach(subject:String) -> String
    func study(subject:String)
}

class ClassTeacher:Teacher{
    var teacherName:String = ""
    var subjects = [String]()
    init(name:String,subjects:[String]){
        self.teacherName = name
        self.subjects = subjects
    }
    func teach(subject:String) -> String {
        return subject
    }
    func study(subject:String ) {
        print("学会了\(subject)！")
        self.subjects.append(subject)
    }
    func say(){
        print("我是\(self.teacherName)，我教\(self.subjects.joined(separator: ","))")
    }
}

var lishi = ClassTeacher(name: "李四", subjects: ["语文","数学"])
lishi.study(subject: "化学")
lishi.say()

extension ClassTeacher{     //拓展类，添加新功能
    func hit(){
        print("\(self.teacherName)，打！打！打！")
    }
}
var zhangsan = ClassTeacher(name: "张三", subjects: ["生物","地理"])
zhangsan.hit()
lishi.hit()
